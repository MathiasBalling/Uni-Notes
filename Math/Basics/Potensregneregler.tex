\[a^r\cdot a^s=a^{r+s}\]

\[\frac{a^r}{a^s}=a^{r-s}\]

\[(a^r)^s=a^{r\cdot s}\]

\[(a\cdot b)^r=a^r\cdot b^r\]

\[\left (\frac{a}{b}\right )^r=\frac{a^r}{b^r}\]

\[a^0=1\]

\[a^{-r}=\frac{1}{a^r}\]

\[a^{-1}=\frac{1}{a}\]

\[\sqrt[r]{a}=a^{\frac{1}{r}}\]

\[\sqrt[s]{a^r}=a^{\frac{r}{s}}\]

\[\sqrt{a\cdot b}=\sqrt{a}\cdot \sqrt{b}\]

\[\sqrt{\frac{a}{b}}=\frac{\sqrt{a}}{\sqrt{b}}\]

\[\sqrt{a}=a^{\frac{1}{2}}\]
